24      public void byDefault_doNotAllowSloppyNames() throws Exception {

30      public void whenNoOptionsAreSpecified_useCurrentDirectoryAsIncludePath() throws Exception {

35      @Test
36      public void whenIncludePathIsSpecified_createIncludeArguments() throws Exception {
37          defineIncludePaths("/src/main/idl-include");
38          mojo.execute();
39          assertArgumentsContains("-I/src/main/idl-include");
40      }

42      @Test
43      public void whenSinglePackagePrefixDefined_createPackageArgument() throws Exception {
44          Source source = createSource();
45          defineSinglePrefix(source, "aPrefix");
46          mojo.execute();
47          assertArgumentsContains("-i2jpackage", ":aPrefix");
48      }
49  
50      @Test
51      public void whenPackagePrefixDefined_createPrefixArguments() throws Exception {
52          Source source = createSource();
53          createPrefix(source, "aType1", "aPrefix1");
54          createPrefix(source, "aType2", "aPrefix2");
55          mojo.execute();
56          assertArgumentsContains("-i2jpackage", "aType1:aPrefix1.aType1");
57          assertArgumentsContains("-i2jpackage", "aType2:aPrefix2.aType2");
58      }

60      @Test
61      public void whenSymbolsDefinedWithoutValues_createSymbolArguments() throws NoSuchFieldException, IllegalAccessException, MojoExecutionException {
62          Source source = createSource();
63          createDefine(source, "symbol1");
64          createDefine(source, "symbol2");
65          mojo.execute();
66          assertArgumentsContains("-Dsymbol1");
67          assertArgumentsContains("-Dsymbol2");

70      @Test
71      public void whenSymbolsDefinedWithValues_createSymbolArguments() throws NoSuchFieldException, IllegalAccessException, MojoExecutionException {
72          Source source = createSource();
73          createDefine(source, "symbol1", "value1");
74          createDefine(source, "symbol2", "value2");
75          mojo.execute();
76          assertArgumentsContains("-Dsymbol1=value1");
